⚡ 구글 시트 쓰기 연동 설정
읽기는 자동 동기화됩니다. 대시보드에서 수정한 내용을 구글 시트에도 저장하려면 아래 Apps Script를 한 번만 설정하면 됩니다.
  1. 1구글 시트를 열고 상단 메뉴 확장 프로그램 → Apps Script 클릭
  2. 2기존 코드를 모두 지우고 아래 코드를 붙여넣기
function doPost(e) {
  try {
    const ss  = SpreadsheetApp.getActiveSpreadsheet();
    const sh  = ss.getSheetByName('주식 수량');
    const dat = sh.getDataRange().getValues();
    const ticker = e.parameter.ticker;
    const field  = e.parameter.field;
    const val    = parseFloat(e.parameter.value);

    for (let i = 1; i < dat.length; i++) {
      if (String(dat[i][0]).trim() === ticker) {
        if (field === '현재가') sh.getRange(i+1, 2).setValue(val);
        if (field === '수량')   sh.getRange(i+1, 3).setValue(val);
        break;
      }
    }
    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,msg:String(err)}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
